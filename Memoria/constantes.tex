%       \titulo
%
% Resultado: 
%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Comando: 
%
%       \autor
%
% Resultado: 
%
% Escribe los autores del documento.
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\def\autor{Pablo Miranda Torres\\Natalia Rodríguez-Peral Valiente\\Jorge Velasco Conde}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Comando: 
%
%       \directores
%
% Resultado: 
%
% Escribe los directores del trabajo de fin de grado.
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\def\directores{Virginia Francisco Gilmartín\\Raquel Hervás Ballesteros}

% Comando: 
%
%       \autorYDirectores
%
% Resultado:

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\def\autorYDirectores{Pablo Miranda Torres\\Natalia Rodríguez-Peral Valiente\\Jorge Velasco Conde\\[1.0em]
	{\Large Directoras}\\[0.2em]

}
